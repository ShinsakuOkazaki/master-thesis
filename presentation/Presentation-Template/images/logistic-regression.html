def update_learning_rate(iteration, hist_cost, hist_lr, bold_driver_en):
    """
    Function Updates the learning rate for the current iteration.
    The updated learning rate is returned via hist_lr[iteration]
    Also takes into account updates with bold driver method
    """
    delta_cost = hist_cost[iteration - 1] - hist_cost[iteration]
    if bold_driver_en:
        if delta_cost < 0:
            hist_lr[iteration] = hist_lr[iteration - 1] * (1 - 10 * bold_drv_factor)
        else:
            hist_lr[iteration] = hist_lr[iteration - 1] * (1 + bold_drv_factor)
    else:
        hist_lr[iteration] = hist_lr[iteration - 1]

def gradient_descent(augmented_rdd, lr_value, max_iter, w_init, b_init, optim=0):
    """
    Computes the cost & gradients
    """
    rdd_gd_input = augmented_rdd.map(lambda x: (x[1], x[2]))
    rdd_gd_input.cache()
    m = rdd_gd_input.count()
    if EN_DBG:
        print("  Gradient descent input format:", rdd_gd_input.first())
    cost_history = [0.0] * max_iter
    lr_history = [0.0] * max_iter;
    lr_history[0] = lr_value
    w = w_init
    b = b_init
    mt = np.empty(w_init.shape[0]); initial_w.fill(0.0)
    vt = np.empty(w_init.shape[0]); initial_w.fill(0.0)
    step = np.empty(w_init.shape[0]); initial_w.fill(0.0)
    beta_1 = 0.9
    beta_2 = 0.999
    epsilon = 0.000001

    def compute_point_gradient(row):
        x = np.array(row[1])
        y = row[0]
        z = np.dot(x, w) + b
        y_hat = 1/(1 + np.exp(-z))
        j = (-1/m)*(y*np.log(y_hat) + (1 - y)*np.log(1 - y_hat))
        d_w = (1/m)*np.dot((y_hat-y), x)
        d_b = (1/m)*(y_hat-y)
        return 'k', (j, d_b, d_w)

    for iteration in range(0, max_iter):
        result = rdd_gd_input.map(compute_point_gradient)
        # print(result.first())
        totals = result.reduceByKey(lambda x, y: (x[0]+y[0], x[1]+y[1], np.add(x[2], y[2]))).collect()[0][1]
        cost_history[iteration] = totals[0]
        dw = totals[2]
        db = totals[1]
        current_w = w
        current_b = b
        
        if iteration > 0:
            if optim==1: # Simple adaptive - bold driver 0.05
                delta_cost = cost_history[iteration - 1] - cost_history[iteration]
                if delta_cost < 0:
                    lr_history[iteration] = lr_history[iteration - 1] * (0.95)
                else:
                    lr_history[iteration] = lr_history[iteration - 1] * (1.05)
            elif optim==2:
                g = np.copy(dw)
                mt = beta_1*mt + (1 - beta_1)*g
                vt = beta_2*vt + (1 - beta_2)*np.power(g, 2)
                mt_hat = mt / (1 - np.power(beta_1, iteration))
                vt_hat = vt / (1 - np.power(beta_2, iteration))
                step = lr_value*mt_hat / (np.sqrt(vt_hat) + epsilon)
            else:
                lr_history[iteration] = lr_history[iteration - 1]
        
        if optim == 2:
            w = w - step
        else:
            w = w - lr_history[iteration]*dw
            b = b - lr_history[iteration]*db
            print("Iteration:", iteration+1, "lr:", lr_history[iteration], "cost:", cost_history[iteration])
        
        if optim == 2: # adam performes different updates to each parameter
            print("Iteration:", iteration+1, "lr:", "cost:", cost_history[iteration])
            
        iteration += 1
    return lr_history, cost_history, b, w

def compute_metrics(corpus_tf):
    rdd_predictions = corpus_tf.map(lambda x: (x[0], x[1], compute_y_prediction(x[2])))
    if EN_DBG:
        print(rdd_predictions.take(10))
    print("\n*Compute model performance metrics...")
    test_metrics = rdd_predictions.map(lambda x: ('k', (x[1], 1 if x[1] == 1 and x[2] == 1 else 0,
                                                              1 if x[1] == 0 and x[2] == 0 else 0,
                                                              1 if x[1] == 0 and x[2] == 1 else 0,
                                                              1 if x[1] == 1 and x[2] == 0 else 0)))\
                                  .reduceByKey(lambda x, y: np.add(x, y)).collect()[0][1]
    metric_TP = test_metrics[1]
    metric_TN = test_metrics[2]
    metric_FP = test_metrics[3]
    metric_FN = test_metrics[4]
    if not(metric_TP + metric_FP) == 0:
        precision = metric_TP/(metric_TP + metric_FP)
    else:
        precision = 1.0
    if not(metric_TP + metric_FN) == 0:
        recall = metric_TP/(metric_TP + metric_FN)
    else:
        recall = 1.0
    F1Score = 2*precision*recall/(precision + recall)
    print(" TruePositives :", metric_TP)
    print(" FalsePositives:", metric_FP)
    print(" TrueNegatives :", metric_TN)
    print(" FalseNegatives:", metric_FN)
    print(" Precision     :", precision)
    print(" Recall        :", recall)
    print(" F1Score       :", F1Score)

def clean_corpus(rdd_raw_corpus):
    """
    Remove invalid rows from the corpus.
    Transforms rdd to format: (label, array[words]) - with the array containing all words in document
    Returns the size of the valid corpus
    """
    reg_exp = re.compile('[^a-zA-Z0-9]')
    raw_corpus_size = rdd_raw_corpus.count()
    print("  Raw corpus size      : ", raw_corpus_size)
    print("  Raw corpus Partitions: ", rdd_raw_corpus.getNumPartitions())
    if EN_DBG:
        print("*Raw corpus Format    : ", rdd_raw_corpus.first())

    rdd_clean_corpus = rdd_raw_corpus.filter(lambda x: "<doc id=" in x and "title=" in x and "</doc>" in x) \
                                     .map(lambda x: (x[x.index('id="') + 4: x.index('" url=')], x[x.index('">') + 2:][:-6])) \
                                     .map(lambda x: (x[0], 1 if "AU" in x[0] else 0, reg_exp.sub(' ', x[1]).lower().split()))
    print("*Removing stop words...")
    rdd_clean_corpus = rdd_clean_corpus.map(lambda x: (x[0], x[1], np.array([wd for wd in x[2] if (wd not in STOP_WORDS_ENG)])))
    valid_corpus_size = rdd_clean_corpus.count()
    print("  Clean corpus size      : ", valid_corpus_size)
    print("  Clean corpus Partitions: ", rdd_clean_corpus.getNumPartitions())
    if EN_DBG:
        print("  Clean corpus Format    : ", rdd_clean_corpus.first())

    return rdd_clean_corpus, valid_corpus_size

def build_corpus_dictionary(rdd_corpus, dict_size):
    """
    Build a dictionary of top <dict_size> high frequency words in the entire corpus
    """
    rdd_word_count = rdd_corpus.flatMap(lambda x: ((wd, 1) for wd in x[2])).reduceByKey(add)
    top_words_count = rdd_word_count.top(dict_size, key=lambda x: x[1])
    top_words = np.array(top_words_count).ravel().reshape(-1, 2)[:, 0]
    assert (len(top_words) == dict_size)
    return top_words

def build_term_frequency(rdd_corpus, bag_of_words_size, corpus_top_words):
    """
    Build the term frequency vector rdd for performing logistic regression
    """
    def sort_term_frequency(row_tf_data):
        tf_row_sorted = np.zeros(bag_of_words_size).astype(np.float)
        term_vec = row_tf_data[0]
        freq_vec = row_tf_data[1].ravel()
        word_count = np.sum(freq_vec.astype(np.float))
        for i in range(0, bag_of_words_size):
            idx = np.where(term_vec == corpus_top_words[i])[0]
            if not (idx.size == 0):
                freq_idx = int(idx)
                tf_row_sorted[i] = freq_vec[freq_idx]
        return tf_row_sorted / word_count

    # Now build the TF vector for each row
    rdd_lbl_tf = rdd_corpus.map(lambda x: (x[0], x[1], np.asarray((np.unique(x[2], return_counts=True))))) \
                           .map(lambda x: (x[0], x[1], sort_term_frequency(x[2][:])))
    if EN_DBG:
        print("*TermFrequency rdd format: ", rdd_lbl_tf.first())
    return rdd_lbl_tf
